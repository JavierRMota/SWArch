def undo
  case @last_speed
  when CeilingFan::HIGH
    @fan.high
  when CeilingFan::MEDIUM
    @fan.medium
  when CeilingFan::LOW
    @fan.low
  when CeilingFan::OFF
    @fan.off
  end
