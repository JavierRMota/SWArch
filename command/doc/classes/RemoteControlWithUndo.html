def inspect
  string_buff = ["\n------ Remote Control -------\n"]
  @on_commands.zip(@off_commands) \
      .each_with_index do |commands, i|
    on_command, off_command = commands
    string_buff << \
    "[slot #{i}] #{on_command.class}  " \
      "#{off_command.class}\n"
  end
  string_buff << "[undo] #{@undo_command.class}\n"
  string_buff.join
end
